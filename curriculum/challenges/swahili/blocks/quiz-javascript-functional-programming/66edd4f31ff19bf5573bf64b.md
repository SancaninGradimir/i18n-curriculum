### --question--

#### --text--
Koji od sledećih predstavlja primer kurijanja?
#### --distractors--

```js
const curriedAverage = (a, b, c) => a + b + c / 3
```

---

```js
const curriedAverage = (a, b, c) => a + b + c
```

---

```js
function curriedAverage(a) {
  return a 
}

function curried(c) {
  return c
}
```

#### --answer--

```js
function curriedAverage(a) {
  return function(b) {
    return function(c) {
      return (a + b + c) / 3;
    };
  };
}
```
